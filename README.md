# Usage : 
Create a MongoDB database and obtain your MongoDB URI - MongoDB Atlas

# Env Variables

NODE_ENV = development
PORT = 5000
MONGO_URI = your mongodb uri

Install Dependencies (frontend & backend)
npm install
cd frontend
npm install

Run

# Run frontend (:3000) & backend (:5000)
npm run dev

# Run backend only
npm run server
